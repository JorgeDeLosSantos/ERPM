Ecuación de la recta que pasa por dos puntos
--------------------------------------------

    x1=input('x1: ');
    y1=input('y1: ');
    x2=input('x2: ');
    y2=input('y2: ');

    b=(-x1*m)+y1;
    if b<0
        signo='- ';
    elseif b>0 
        signo='+ ';
    else
        signo='';
        b='';
    end

    end
